-- INNER JOINE
SELECT
    E.F_name || ' ' || E.L_name AS Employee_Name,
    D.D_name AS Department_Name,
    P.P_Name AS Project_Name
FROM EMPLOYEES E
INNER JOIN DEPARTMENT D
    ON E.D_NUM = D.D_NUM
INNER JOIN EMPLOYEE_WORKING_ON_PROJECT W
    ON E.SSN = W.SSN
INNER JOIN PROJECTS P
    ON W.P_NUM = P.P_NUM;


-- LEFT JOINE
SELECT
    E.F_name || ' ' || E.L_name AS Employee_Name,
    D.D_name AS Department_Name,
    DP.D_Name AS Dependent_Name,
    DP.Gender
FROM EMPLOYEES E
LEFT JOIN DEPARTMENT D
    ON E.D_NUM = D.D_NUM
LEFT JOIN DEPENDENTS DP
    ON E.SSN = DP.SSN_employee;
    
-- RIGHT JOINE
SELECT
    P.P_Name AS Project_Name,
    D.D_name AS Department_Name,
    E.F_name || ' ' || E.L_name AS Employee_Name
FROM EMPLOYEES E
RIGHT JOIN DEPARTMENT D
    ON E.D_NUM = D.D_NUM
RIGHT JOIN PROJECTS P
    ON D.D_NUM = P.D_NUM;
    
-- SELF JOINE
SELECT
    E.F_name || ' ' || E.L_name AS Employee_Name,
    S.F_name || ' ' || S.L_name AS Supervisor_Name,
    D.D_name AS Department_Name
FROM EMPLOYEES E
LEFT JOIN EMPLOYEES S
    ON E.SSN_supervisor = S.SSN
LEFT JOIN DEPARTMENT D
    ON E.D_NUM = D.D_NUM;
