CREATE TABLE EMPLOYEES(
    SSN NUMBER(10) PRIMARY KEY NOT NULL,
    F_name VARCHAR(20) NOT NULL,
    L_name VARCHAR(20) NOT NULL,
    Birth_Date DATE NOT NULL,
    Gender VARCHAR(10) NOT NULL,
    SSN_supervisor NUMBER(10),
    FOREIGN KEY (SSN_supervisor) REFERENCES EMPLOYEES(SSN)
); 

CREATE TABLE DEPARTMENT (
    D_NUM NUMBER(10) PRIMARY KEY NOT NULL,
    D_name VARCHAR(20) NOT NULL,
    SSN_manager NUMBER(10),
    FOREIGN KEY (SSN_manager) REFERENCES EMPLOYEES(SSN)
);

CREATE TABLE DEPENDENTS(
    D_Name VARCHAR(30) PRIMARY KEY NOT NULL,
    Gender VARCHAR(10) NOT NULL,
    Birth_Name DATE NOT NULL,
    SSN_employee NUMBER(10),
    FOREIGN KEY (SSN_employee) REFERENCES EMPLOYEES(SSN)
);

CREATE TABLE PROJECTS (
    P_NUM NUMBER(10) PRIMARY KEY NOT NULL,
    P_Name VARCHAR(20) NOT NULL,
    Location VARCHAR(20) NOT NULL,
    D_NUM NUMBER (10),
    FOREIGN KEY (D_NUM) REFERENCES DEPARTMENT(D_NUM)
);

CREATE TABLE EMPLOYEE_WORKING_ON_PROJECT(
    SSN NUMBER(10),
    P_NUM NUMBER(10),
    FOREIGN KEY (SSN) REFERENCES EMPLOYEES(SSN),
    FOREIGN KEY (P_NUM) REFERENCES PROJECTS(P_NUM),
    Working_hours NUMBER(5) NOT NULL,
    PRIMARY KEY (SSN, P_NUM)
);

CREATE TABLE DEPARTMENT_LOCATION(
    D_NUM NUMBER(10),
    FOREIGN KEY (D_NUM) REFERENCES DEPARTMENT(D_NUM),
    Locations VARCHAR(20) NOT NULL,
    PRIMARY KEY (D_NUM, Locations)
);
